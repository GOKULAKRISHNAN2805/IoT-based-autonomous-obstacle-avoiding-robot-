// ========================================================

const char* password = "YOUR_WIFI_PASSWORD";

// ========================================================
// ESP32-CAM Stream URL
// ========================================================

#define CAM_IP "http://YOUR_ESP32_CAM_IP/"

// ========================================================
// Web Server
// ========================================================

ESP8266WebServer server(80);

// ========================================================
// Motor Pins
// ========================================================

#define IN1 D1
#define IN2 D2
#define IN3 D3
#define IN4 D4

// ========================================================
// Ultrasonic Sensor Pins
// ========================================================

#define TRIG D5
#define ECHO D6

// ========================================================
// Servo Motor
// ========================================================

Servo servo;

#define SERVO_PIN D7

int center = 90;
int leftAngle = 180;
int rightAngle = 0;

// ========================================================
// Global Variables
// ========================================================

bool autoMode = false;

int motorSpeed = 800;

// ========================================================
// Distance Measurement
// ========================================================

long getDistance() {

  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);

  digitalWrite(TRIG, LOW);

  long duration = pulseIn(ECHO, HIGH, 30000);

  if (duration == 0) {
    return 999;
  }

  return duration * 0.034 / 2;
}

// ========================================================
// Motor Functions
// ========================================================

void forward() {

  servo.write(center);

  analogWrite(IN1, motorSpeed);
  analogWrite(IN2, 0);

  analogWrite(IN3, motorSpeed);
  analogWrite(IN4, 0);
}

void backward() {

  analogWrite(IN1, 0);
  analogWrite(IN2, motorSpeed);

  analogWrite(IN3, 0);
  analogWrite(IN4, motorSpeed);
}

void stopM() {

  analogWrite(IN1, 0);
  analogWrite(IN2, 0);

  analogWrite(IN3, 0);
  analogWrite(IN4, 0);
}

// ========================================================
// Turning Functions
// ========================================================

void turnLeft90() {

  analogWrite(IN1, 0);
  analogWrite(IN2, motorSpeed);

  analogWrite(IN3, motorSpeed);
  analogWrite(IN4, 0);

  delay(400);

  stopM();
}

void turnRight90() {

  analogWrite(IN1, motorSpeed);
  analogWrite(IN2, 0);

  analogWrite(IN3, 0);
  analogWrite(IN4, motorSpeed);

  delay(400);

  stopM();
}

// ========================================================
// Autonomous Navigation
// ========================================================

void autoRun() {

  long front = getDistance();

  if (front < 40) {

    stopM();

    delay(100);

    backward();

    delay(300);

    stopM();

    // Scan Left

    servo.write(leftAngle);

    delay(400);

    long leftDist = getDistance();

    // Scan Right

    servo.write(rightAngle);

    delay(400);

    long rightDist = getDistance();

    // Return Center

    servo.write(center);

    delay(200);

    // Decision Making

    if (leftDist < 25 && rightDist < 25) {

      backward();

      delay(500);

      stopM();

      if (leftDist > rightDist) {

        turnLeft90();

      } else {

        turnRight90();
      }

    } else if (leftDist < rightDist) {

      turnRight90();

    } else {

      turnLeft90();
    }

  } else {

    forward();
  }
}

// ========================================================
// Speed Control
// ========================================================

void handleSpeed() {

  if (server.hasArg("val")) {

    motorSpeed = server.arg("val").toInt();

    if (motorSpeed < 0) {
      motorSpeed = 0;
    }

    if (motorSpeed > 1023) {
      motorSpeed = 1023;
    }
  }

  server.send(200, "text/plain", "OK");
}

// ========================================================
// Send Robot Data
// ========================================================

void handleData() {

  long d = getDistance();

  String json =
    "{\"d\":" + String(d) +
    ",\"m\":\"" + String(autoMode ? "Auto" : "Manual") +
    "\",\"s\":" + String(motorSpeed) + "}";

  server.send(200, "application/json", json);
}

// ========================================================
// Web Handlers
// ========================================================

void handleRoot() {

  String page = html;

  page.replace("%CAMIP%", CAM_IP);

  server.send(200, "text/html", page);
}

void handleF() {

  if (!autoMode) {

    long d = getDistance();

    if (d < 30) {
      stopM();
    } else {
      forward();
    }
  }

  server.send(200, "text/plain", "OK");
}

void handleB() {

  if (!autoMode) {
    backward();
  }

  server.send(200, "text/plain", "OK");
}

void handleL() {

  if (!autoMode) {

    long d = getDistance();

    if (d < 30) {
      stopM();
    } else {
      turnLeft90();
    }
  }

  server.send(200, "text/plain", "OK");
}

void handleR() {

  if (!autoMode) {

    long d = getDistance();

    if (d < 30) {
      stopM();
    } else {
      turnRight90();
    }
  }

  server.send(200, "text/plain", "OK");
}

void handleS() {

  stopM();

  server.send(200, "text/plain", "OK");
}

void handleMode() {

  autoMode = !autoMode;

  stopM();

  server.send(200, "text/plain", "OK");
}

// ========================================================
// Setup
// ========================================================

void setup() {

  Serial.begin(115200);

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);

  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  pinMode(TRIG, OUTPUT);

  pinMode(ECHO, INPUT);

  servo.attach(SERVO_PIN, 500, 2400);

  servo.write(center);

  // WiFi Connection

  WiFi.begin(ssid, password);

  Serial.print("Connecting to WiFi");

  while (WiFi.status() != WL_CONNECTED) {

    delay(500);

    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi Connected");

  Serial.println(WiFi.localIP());

  // Server Routes

  server.on("/", handleRoot);

  server.on("/f", handleF);
  server.on("/b", handleB);

  server.on("/l", handleL);
  server.on("/r", handleR);

  server.on("/s", handleS);

  server.on("/mode", handleMode);

  server.on("/speed", handleSpeed);

  server.on("/data", handleData);

  // Start Server

  server.begin();

  Serial.println("Web Server Started");
}

// ========================================================
// Main Loop
// ========================================================

void loop() {

  server.handleClient();

  if (!autoMode) {

    servo.write(center);

    long d = getDistance();

    if (d < 30) {

      stopM();
    }
  }

  if (autoMode) {

    autoRun();

    servo.write(center);
  }
}
